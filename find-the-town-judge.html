/**
* @param {number} n
* @param {number[][]} trust
* @return {number}
*/
var findJudge = function(n, trust) {
    if (trust.length === 0 && n === 1) return 1

    const ins = []
    const outs = []

    trust.forEach(([from, to]) => {
        outs[from] = (outs[from] || 0) + 1
        ins[to] = (ins[to] || 0) + 1
    })

    for (let i = 1; i <= n; i += 1) {
        if (ins[i] === n - 1 && !outs[i]) {
            return i
        }
    }

    return -1
};
