/**
* @param {string} input
* @returns string
*/
function removeChars(input) {
  const {length} = input
  const stack = new Stack
  for (let i = 0; i < length; i += 1) {
    switch (input[i]) {
      case 'b':
        continue;
      case 'c':
        if (stack.peek() === 'a') {
          stack.pop()
          continue;
        }
      default:
        stack.push(input[i])
        continue;
    }
  }

  return stack.toString()
}

class Stack {
  constructor() {
    this.list = ''
  }
  peek() {
    const {length} = this.list
    if (length > 0) return this.list[length - 1]
    return -1
  }
  pop() {
    this.list = this.list.slice(0, -1)
  }
  push(v) {
    this.list += v
  }
  toString() {
    return this.list
  }
}
