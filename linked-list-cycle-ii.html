/**
* Definition for singly-linked list.
* function ListNode(val) {
*     this.val = val;
*     this.next = null;
* }
*/

/**
* @param {ListNode} head
* @return {ListNode}
*/
var detectCycle = function(head) {
    if (!head) return null
    let fast = head
    let slow = head

    while (fast) {
        slow = slow.next
        if (!fast.next || !fast.next.next) {
            return null
        }
        fast = fast.next.next

        if (fast === slow) {
            fast = head
            while (fast !== slow) {
                fast = fast.next
                slow = slow.next
            }
            return fast
        }
    }
};
