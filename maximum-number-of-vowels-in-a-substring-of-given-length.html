/**
* @param {string} s
* @param {number} k
* @return {number}
*/
var maxVowels = function(s, k) {
    const { length } = s
    let max = 0
    let count = 0
    for (let i = 0; i < k - 1; i += 1) {
        if (isVowel(s[i])) {
            count += 1
        }
    }
    for (let i = k - 1; i < length; i += 1) {
        if (isVowel(s[i])) {
            count += 1
        }
        max = Math.max(max, count)
        if (isVowel(s[i - k + 1])) {
            count -= 1
        }
    }

    return max
};

function isVowel(char) {
    return 'aeiou'.includes(char)
}
