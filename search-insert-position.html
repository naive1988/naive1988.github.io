/**
* @param {number[]} nums
* @param {number} target
* @return {number}
*/
var searchInsert = function(nums, target) {
    if (nums.length === 1) {
        return target > nums[0] ? 1 : 0
    }
    const idx = Math.floor(nums.length / 2)

    if (target < nums[idx]) {
        return searchInsert(nums.slice(0, idx), target)
    }
    return idx + searchInsert(nums.slice(idx), target)
};
